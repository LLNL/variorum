  ${CMAKE_CURRENT_SOURCE_DIR}/Power9.h
  ${CMAKE_CURRENT_SOURCE_DIR}/ibm_sensors.h

  ${CMAKE_CURRENT_SOURCE_DIR}/Power9.c
